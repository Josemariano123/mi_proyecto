class Conexion {
    public static function conexionBD() {
        $host = "localhost";
        $dbname = "escuela"; // CAMBIA si tu BD tiene otro nombre
        $user = "root";
        $pass = "";

        try {
            $conexion = new PDO("mysql:host=$host;dbname=$dbname;charset=utf8", $user, $pass);
            $conexion->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
            return $conexion;
        } catch (PDOException $e) {
            echo "<p class='text-red-500'>Error de conexión: " . $e->getMessage() . "</p>";
            return null;
        }
    }
}

// Intentar conexión
